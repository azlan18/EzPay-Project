## Setup
-- 1. Clone the repo
-- 2. cd EzPay-Project/backend
      npm i
      cd ../frontend
      npm i
-- 3. Add your mongodb string at /backend/.env
-- 4. Add your own JWT_SECRET at /backend/config.js
-- 4. Start the backend server
      cd backend
      node index.js
-- 5. Start the client server
      cd ../frontend 
      npm run dev
